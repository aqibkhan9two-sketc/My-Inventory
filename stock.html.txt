    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-app.js";
        import { getFirestore, collection, onSnapshot } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-firestore.js";

        const firebaseConfig = {
            apiKey: "AIzaSyBTxU9U4OW_xaEV3WZnSrpdM9Lal8TBN5c",
            authDomain: "my-inventory-app-4c806.firebaseapp.com",
            projectId: "my-inventory-app-4c806",
            storageBucket: "my-inventory-app-4c806.firebasestorage.app",
            messagingSenderId: "258251635272",
            appId: "1:258251635272:web:597ea77880c4ff3ef3767d"
        };

        const app = initializeApp(firebaseConfig);
        const db = getFirestore(app);

        // Filter Function without touching data logic
        window.filterStockTable = () => {
            const searchVal = document.getElementById('stockSearch').value.toUpperCase();
            const prefixVal = document.getElementById('prefixFilter').value.toUpperCase();
            const rows = document.querySelectorAll('#stockTableBody tr');

            rows.forEach(row => {
                const skuText = row.cells[0].innerText.toUpperCase();
                const nameText = row.cells[1].innerText.toUpperCase();

                const matchesSearch = skuText.includes(searchVal) || nameText.includes(searchVal);
                const matchesPrefix = prefixVal === "" || skuText.startsWith(prefixVal);

                row.style.display = (matchesSearch && matchesPrefix) ? "" : "none";
            });
        };

        function loadStockData() {
            onSnapshot(collection(db, 'warehouses'), (whSnap) => {
                const warehouses = whSnap.docs.map(d => d.data().name);

                // Setup Prefix Dropdown from SKU list (optional but helpful)
                // For now, we'll keep it simple as per your request.

                let headerHTML = `
                <th>SKU</th>
                <th>Product Name</th>
                <th class="text-center">Purchased (Total)</th>
                <th class="text-center">Orders (Total)</th>
                <th class="text-center">Balance (Live)</th>
                <th class="text-center">Avg Cost (Unit)</th>`;

                warehouses.forEach(wh => {
                    headerHTML += `<th class="wh-col text-center">${wh.toUpperCase()}</th>`;
                });
                headerHTML += `<th>Product Link</th>`;

                const thead = document.querySelector('thead tr');
                if (thead) thead.innerHTML = headerHTML;

                onSnapshot(collection(db, 'products'), (prodSnap) => {
                    // Populate Prefix Filter
                    const prefixes = new Set();
                    prodSnap.docs.forEach(d => {
                        const sku = d.data().sku || "";
                        if (sku.includes('-')) prefixes.add(sku.split('-')[0]);
                    });
                    let prefOptions = '<option value="">All Prefixes</option>';
                    prefixes.forEach(p => prefOptions += `<option value="${p}">${p}</option>`);
                    document.getElementById('prefixFilter').innerHTML = prefOptions;

                    onSnapshot(collection(db, 'purchases'), (purSnap) => {
                        onSnapshot(collection(db, 'orders'), (ordSnap) => {
                            const allPurchases = purSnap.docs.map(d => d.data());
                            const allOrders = ordSnap.docs ? ordSnap.docs.map(d => d.data()) : [];
                            let rows = '';

                            prodSnap.docs.forEach(doc => {
                                const product = doc.data();
                                const sku = product.sku;
                                const skuPurchases = allPurchases.filter(p => p.sku === sku);

                                const totalQtyIn = skuPurchases.reduce((sum, p) => sum + (+p.qty || 0), 0);
                                const totalSpend = skuPurchases.reduce((sum, p) => sum + (+p.totalAmount || 0), 0);
                                const totalQtyOut = allOrders.filter(o => o.sku === sku).reduce((sum, o) => sum + (+o.qty || 0), 0);

                                const balance = totalQtyIn - totalQtyOut;
                                const avgCost = totalQtyIn > 0 ? (totalSpend / totalQtyIn).toFixed(2) : "0.00";

                                let whColumnsHTML = '';
                                warehouses.forEach(whName => {
                                    const whIn = skuPurchases.filter(p => p.warehouse === whName).reduce((sum, p) => sum + (+p.qty || 0), 0);
                                    const whOut = allOrders.filter(o => o.sku === sku && o.warehouse === whName).reduce((sum, o) => sum + (+o.qty || 0), 0);
                                    const whFinalBalance = whIn - whOut;
                                    whColumnsHTML += `<td class="wh-col text-center">${whFinalBalance}</td>`;
                                });

                                rows += `
                                <tr>
                                    <td><strong>${sku}</strong></td>
                                    <td>${product.name}</td>
                                    <td class="text-center"><span class="qty-box purchased">${totalQtyIn}</span></td>
                                    <td class="text-center"><span class="qty-box orders">${totalQtyOut}</span></td>
                                    <td class="text-center"><span class="qty-box balance">${balance}</span></td>
                                    <td class="avg-cost text-center">$${avgCost}</td>
                                    ${whColumnsHTML}
                                    <td><a href="${product.link || '#'}" target="_blank" style="color:#3498db; text-decoration:none;">Open Link</a></td>
                                </tr>`;
                            });

                            document.getElementById('stockTableBody').innerHTML = rows || '<tr><td colspan="20" style="text-align:center;">No inventory data found.</td></tr>';
                            document.getElementById('lastSync').innerText = "Last Updated: " + new Date().toLocaleTimeString();

                            // Re-apply filters after data update
                            filterStockTable();
                        });
                    });
                });
            });
        }

        loadStockData();
    </script>
